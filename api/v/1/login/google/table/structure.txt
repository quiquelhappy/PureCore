CREATE TABLE `purecore_data`.`accounts` ( `coreid` VARCHAR(16) NOT NULL , `email` VARCHAR(256) NULL , `name` VARCHAR(256) NULL , `surname` VARCHAR(256) NULL , `uuid` VARCHAR(36) NOT NULL , `username` VARCHAR(16) NOT NULL , `joined` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `servers` LONGTEXT NOT NULL , `hash` LONGTEXT NOT NULL , `corehash` VARCHAR(32) NOT NULL , `ip` VARCHAR(32) NOT NULL , `country` VARCHAR(2) NOT NULL , `state` VARCHAR(128) NOT NULL , `id` BIGINT NOT NULL AUTO_INCREMENT , PRIMARY KEY (`coreid`), UNIQUE (`id`)) ENGINE = InnoDB;
