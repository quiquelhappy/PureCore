CREATE TABLE `purecore_data`.`sessions` ( `id` VARCHAR(16) NOT NULL , `owner` VARCHAR(16) NULL DEFAULT NULL , `device` VARCHAR(256) NULL DEFAULT NULL , `platform` VARCHAR(256) NULL DEFAULT NULL , `browser` VARCHAR(256) NULL DEFAULT NULL , `languages` TEXT NULL DEFAULT NULL , `location` TEXT NULL DEFAULT NULL , `hash` VARCHAR(128) NOT NULL , `account` VARCHAR(16) NULL DEFAULT NULL , `creation` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `active` BOOLEAN NOT NULL DEFAULT TRUE , `ip` VARCHAR(32) NOT NULL ) ENGINE = InnoDB;
